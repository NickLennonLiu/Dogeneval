操作场景

部署基于域名的本地分流功能，且本地DN部署DNS服务器时，需要配置七层分流规则和DNS重定向。

必备事项

前提条件

  • 请仔细阅读分流原理。
  • 已完成与周边网元的对接。
  • 完成协议识别库和解析库的加载。
  • 已开启Web Proxy功能。

操作步骤

  1. 配置UL CL与PSA1共部署的本地分流功能开关。

    SET COLOCATEDLBO

  2. 配置Host信息。

    ADD HOST

  3. 配置基于域名的本地分流过滤条件。
    1. 配置流过滤器。

      ADD FLOWFILTER

    2. 配置三四层过滤条件。

      ADD FILTER

    3. 绑定三四层过滤条件。

      ADD FLTBINDFLOWF

    4. 配置七层过滤条件。

      ADD L7FILTER

    5. 绑定七层过滤条件。

      ADD PROTBINDFLOWF

  4. 配置本地分流流过滤器组。

    ADD FLOWFILTERGRP

    • 流过滤器组对应SMF下发的AppID。一个流过滤器组中需要绑定基于域名的本地分流过滤条件，才能实现针对一个AppID的分流功能。
    • 匹配中流过滤器组后，根据ApplD检索到分流PDR，然后按照PDR中的转发规则FAR、QoS规则QER执行策略动作。

  5. 配置IP Farm使用的心跳检测接口。

    ADD LOGICINF

  6. 配置IP Farm。
    1. 创建重定向的IP Farm。

      ADD IPFARM

    2. 配置IP farm服务器地址。

      ADD IPFARMSERVER

      IP Farm配置完成后，可以执行命令DSP IPFARMSERVER查询IP Farm中server的状态。为确保业务正常进行，需保证至少有一个server的状态为up。

  7. 配置重定向规则。

    ADD RULE

  8. 配置重定向规则与用户模板的绑定关系。
    1. 创建UserProfile。

      ADD USERPROFILE

    2. 将已配置的rule绑定到UserProfile。

      ADD RULEBINDING

任务示例

任务描述

开启UPF本地分流功能，UPF将访问URL为www.huawei.com的DNS报文重定向到地址为"192.168.253.251"的本地DNS服务器。

脚本

//配置UL CL与PSA1共部署的本地分流功能开关。

SET COLOCATEDLBO: ULCLSWITCH=ENABLE;

//配置Host信息。

ADD HOST:HOSTNAME="host1",DOMAINVALUE="www.huawei.com";

//配置UL CL UPF的针对DNS报文的L7过滤条件。

ADD FLOWFILTER:FLOWFILTERNAME="flowfilter_ulcl01";
ADD FILTER:FILTERNAME="filter_test01",L34PROTTYPE=STRING,L34PROTOCOL=UDP,SVRSTARTPORT=53,SVRENDPORT=53;
ADD FLTBINDFLOWF:FLOWFILTERNAME="flowfilter_ulcl01",FILTERNAME="filter_test01";
ADD L7FILTER:L7FILTERNAME="l7filter-test",SUBL7FLTNAME="sl7filter-test",URLTYPE=URL,URL="www.huawei.com";
ADD PROTBINDFLOWF:FLOWFILTERNAME="flowfilter_ulcl01",PROTOCOLNAME="dns",L7FILTERNAME="l7filter-test";

//配置UL CL UPF的针对IP报文的L34过滤条件。

ADD FLOWFILTER:FLOWFILTERNAME="flowfilter_ulcl02";
ADD FILTER:FILTERNAME="filter_test02",L34PROTTYPE=STRING,L34PROTOCOL=ANY,SVRIPMODE=HOSTNAME,HOSTNAME="host1";
ADD FLTBINDFLOWF:FLOWFILTERNAME="flowfilter_ulcl02",FILTERNAME="filter_test02";

//配置UL CL UPF的本地分流流过滤器组。

ADD FLOWFILTERGRP:FLWFLTRGRPNAME="flowfiltergrp_ulcl01",FLOWFILTERNAME="flowfilter_ulcl01",FLWFLTRLOGICGRP=OR;
ADD FLOWFILTERGRP:FLWFLTRGRPNAME="flowfiltergrp_ulcl01",FLOWFILTERNAME="flowfilter_ulcl02",FLWFLTRLOGICGRP=OR;

//配置IP Farm使用的心跳检测接口。

ADD LOGICINF:NAME="phif1/0/0",IPVERSION=IPV4,IPV4ADDRESS1="10.10.10.10",IPV4MASK1="255.255.255.255";
//配置PSA1 UPF的IP Farm。
ADD IPFARM:HEALTHCHECKFLAG=ENABLE,IPFARMNAME="farm_test",IPVERSION=IPV4,INTERFACENAME="phif1/0/0";
ADD IPFARMSERVER:IPFARMNAME="farm_test",IPVERSION=IPV4,SERVERIPV4="192.168.253.251";

//配置PSA1 UPF的重定向规则。

ADD RULE:RULENAME="rule_dns01",POLICYTYPE=WEBPROXY,FILTERINGMODE=FLOWFILTER, FLOWFILTERNAME="flowfilter_ulcl01",POLICYNAME="farm_test";

//配置PSA1 UPF的重定向规则与用户模板的绑定关系。

ADD USERPROFILE:USERPROFILENAME="up_test";
ADD RULEBINDING:USERPROFILENAME="up_test",RULENAME="rule_dns01";
