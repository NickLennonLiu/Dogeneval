操作场景

为了避免NF的异常操作影响到NRF设备安全及5G Core业务，NRF支持对NF进行白名单和校验控制：符合NRF白名单的NF才允许注册/更新；NF进行上述服务化流程时，对应参数校验通过的NF才能完成对应的服务化流程。

为了避免号段类NF的号段配置错误影响到5G Core业务，NRF支持对NF Profile的属性冲突核验，对于存在号段冲突的情况可以查询或提示告警。

必备事项

前提条件

  • 具备管理员权限的操作人员已经登录华为操作维护系统（登录U2020/MAE客户端或登录OM Portal）。

  • 完成加载License。
  • NRF解决方案基本功能在NF侧和NRF侧均已配置完成，详细参考NRF解决方案基本功能配置实例（NF注册/更新/去注册、NF订阅/通知）。

操作步骤

    任务场景1：属性冲突检查等可靠性功能。为了确保NF Profile正确性，避免NF由于属性配置错误导致NF间存在属性冲突，AMF和NRF侧进行如下可靠性配置：

    //打开属性冲突检查相关License。
SET LICENSESWITCH: LICITEM="LKV2NFACV01", SWITCH=ENABLE;

//白名单接入控制功能。
//NRF侧打开NF白名单接入控制功能。NRF将处理白名单内NF的注册、更新及心跳，未在白名单内的NF将无法正常注册、更新以及维持到NRF的心跳。
SET NRFWHITELISTSW: NFWHITELISTSW=FUNC_ON;
//AMF侧人工检验NF Profile符合规范后，NRF侧将此AMF的InstanceID：88888888-4444-1234-5678-123456789ABC加入到白名单内，确保可以到NRF上注册/更新。 
ADD NRFWHITELIST: NFINSTANCEID="88888888-4444-1234-5678-123456789ABC";

//AMF注册必须携带TAI List或TAI Range中的一个参数，避免NRF认为不带TAI相关信息的AMF支持所有TAI而选错AMF。
//AMF侧检查此规则。
SET NFNRFMGMTPARA: NFPROFILECHKSW=AMF_TAI-1;
//NRF侧检查此规则。
SET NRFTAKEPARARULE: NFTYPE=AMF, NOTAISW=FUNC_ON;

//NRF侧处理携带array下标的部分更新流程为更新失败，避免NF和NRF间对array下标处理不一致而影响业务。
SET NRFPATUPDIDXSW: PATUPDWITHIDXSW=FUNC_ON;

//NRF侧配置，目标为AMF的服务发现，发现参数除基本发现参数及NF Type外，必须携带其他发现参数，否则服务发现失败。
SET NRFDISCPARARULE: NFTYPE=AMF, DISCPARA=TARGETNFTYPE-1;

//对AMF的NF Profile的属性冲突进行人工核验。
//NRF侧对本NRF管理的AMF间TAI属性进行核验，选择NF实例标识中大区省份信息起始位置为4信息长度为3的部分。
OPR NRFNFATTRVRY: NFINSTANCEID="88888888-4444-1234-5678-123456789ABC"; 
SET NRFVERIFYPARA: START=4, LENGTH=3, VRYLOCALATTR=TAI-1;
//可以通过DSP NRFNFATTRVRY查看核验结果。
DSP NRFNFATTRVRY: NFINSTANCEID="88888888-4444-1234-5678-123456789ABC";

//对AMF的NF Profile的属性冲突进行自动核验。
//NRF侧对本NRF管理服务发现结果的NF进行核验，选择NF实例标识中大区省份信息起始位置为4信息长度为3的部分，如果存在NF与当前NF的实现大区省份信息不一致会核验上报告警。对于跨NRF路由属性冲突也进行核验，如果本NRF服务发现结果有匹配的AMF，跨NRF寻址路由也有符合的NF，则会核验上报告警。
SET NRFDISCVERIFY: DISCVRYFSW=FUNC_ON, START=4, LENGTH=3, DISCLOCALATTR=TAI-1, DISCINTERATTR=TAI-1;
//AMF侧打开NF实例标识冲突核查开关。AMF在本地缓存中匹配成功的NF存在大区/省份标识不同，则上报告警。
SET NFDISCPLCY: NFIDCHECK=ON;

    任务场景2：为了避免NF和NRF间传递大包影响网络带宽，NF和NRF侧设置各自的最大包处理能力。

    //AMF和NRF上分别设置最大包处理能力，保证网络稳定性，避免超大包影响网络带宽。
//NRF侧配置支持单NF的最大号段数量20000，服务发现返回的最大号段数40000，最大报文长度200千字节。
SET NRFBIGPKGPARA: MAXREGSEGNUM=20000, MAXDISCSEGNUM=40000, MAXPKGSIZE=200;
//AMF侧设置最大有效负载，在服务发现过程中带给NRF，告知返回NF profile的最大字节数为160千字节。
SET NFDISCPLCY: MAXPLDSIZE=160;
